//Receber um texto de até 100 caracteres contendo letras e números e contar quantas vezes aparece qualquer letra

Algoritmo "ContarCaracteres"
Var
    Texto: caractere
    i, contador, n: inteiro
    lista: [contem espaço e acentos e nulo]
Inicio
    contador = 0

    Escreva("Digite o texto: ")
    Leia(Texto)

    Para i de 1 até comprimento(Texto) faça
        para n de 1 a comprimento(lista) faça 
            Se Texto[i] != lista[n] então
                contador++
            FimSe
        FimPara
    FimPara

    Escreva("Quantidade de letras: ", contador)

FimAlgoritmo
